{{/* Renders a single tag chip. Argument is EITHER: - a "facet.value" string -> a (safe inside card links, no nested ) - a dict {value, link:true} -> an linking to the /tags// term page Splits on the first "." to get facet and value; strips the prefix for display. Falls back to the full string if no "." is found. Matching is case-insensitive so it works both for original-case tags in front matter and the lowercased terms Hugo emits on taxonomy pages. Color comes from .tag--{facet}. The link URL is built from the lowercased tag (Hugo lowercases taxonomy terms) run through relURL so it carries the /hugo2/ base path — checked by scripts/check_links.py. */}} {{ $raw := . }}{{ $link := false }} {{ if reflect.IsMap . }}{{ $raw = .value }}{{ $link = .link }}{{ end }} {{ $parts := split $raw "." }} {{ if ge (len $parts) 2 }} {{ $facet := index $parts 0 }} {{ $value := strings.TrimPrefix (printf "%s." $facet) $raw }} {{ $label := cond (findRE `^\d+$` $value) $value (humanize $value) }} {{/* look up friendly label from the canonical registry: values first, then aliases */}} {{ range hugo.Data.tags }} {{ if eq (lower .facet) (lower $facet) }} {{ range .values }} {{ if eq (lower .value) (lower $value) }}{{ $label = .name }}{{ end }} {{ end }} {{ range $k, $v := .aliases }} {{ if eq (lower $k) (lower $value) }}{{ $label = $v }}{{ end }} {{ end }} {{ end }} {{ end }} {{ if $link }}{{ $label }} {{ else }}{{ $label }}{{ end }} {{ else }} {{ if $link }}{{ humanize $raw }} {{ else }}{{ humanize $raw }}{{ end }} {{ end }}
