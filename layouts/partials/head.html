{{- /* Bundle every @import listed in styles.css into one fingerprinted file, so the browser cache busts automatically whenever any CSS changes. styles.css stays the single source of cascade order. */ -}} {{- $manifest := readFile "static/css/styles.css" -}} {{- $css := "" -}} {{- range findRE `@import "[^"]+"` $manifest -}} {{- $rel := replaceRE `@import "([^"]+)"` "$1" . -}} {{- $css = printf "%s\n%s" $css (readFile (printf "static/css/%s" $rel)) -}} {{- end -}} {{- $styles := resources.FromString "css/styles.css" $css | fingerprint -}}
